algoritmo "Temperatura Graus Celsius para Fahrenheit"

var
     C, F  : real

inicio

  //entrada de dados
   escreva ("Graus Celsius: ")
   leia (C)

   // processamento
    F <- (9 * C + 160) / 5
    
    // saida de dados
    escreva ("Fahrenheit=  ", F )
